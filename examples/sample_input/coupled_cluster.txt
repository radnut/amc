# Coupled cluster energy
declare DE {
    mode=0,
    latex="{\Delta E_{CC}}",
}

# Coupled cluster amplitudes
declare T1 {
    mode=2,
    latex="t",
    reduce=true
}
declare T2 {
    mode=4,
    latex="t"
}

# Zero tensor for external coupling
declare ZERO {
    mode=4,
    latex="0"
}

# Fock operator
declare F {
    mode=2,
    latex="f",
    reduce=true
}

# Two-body Hamiltonian
declare H {
    mode=4,
    latex="H",
}

# Energy equation
DE = sum_ai(F_ia * T1_ai)
   + sum_abij(H_ijab * T1_ai * T1_bj)
   + 1/4 * sum_abij(H_ijab * T2_abij);

# One term of CCSD amplitude equations
ZERO_abij = sum_klcd(H_klcd * T1_dj * T1_ak * T2_cbil);
